int motor1Pin1 = 4;		// pin 2 on L293D
int motor1Pin2 = 5; 	// pin 7 on L293D
int enablePin1 = 3;		// pin 1 on L293D
int motor2Pin1 = 8;		// pin 10 on L293D
int motor2Pin2 = 9;		// pin 15 on L293D
int enablePin2 = 10;	// pin 9 on L293D
int RX0 = 0;			// pin RX0 on Bluetooth
int TX0 = 1;			// pin TX0 on Bluetooth

void setup() {

  // These are the motor pins, they are outputs:
  pinMode(motor1Pin1, OUTPUT);
  pinMode(motor1Pin2, OUTPUT);
  pinMode(enablePin1, OUTPUT);

  // Set enablePin high so that the motors can turn on:
  digitalWrite(enablePin1, HIGH);
  digitalWrite(enablePin2, HIGH);
}

void loop() {
  // If the motor in enabled then it will turn:
  if (digitalWrite(enablePin1) == HIGH) {
    digitalWrite(motor1Pin1, LOW);   // set pin 2 on L293D low
    digitalWrite(motor1Pin2, HIGH);  // set pin 7 on L293D high
	digitalWrite(motor2Pin1, LOW);   // set pin 10 on L293D low
    digitalWrite(motor2Pin2, HIGH);  // set pin 15 on L293D high
  }
  // if the switch is low, motor will turn in the opposite direction:
  else {
    digitalWrite(motor1Pin1, HIGH);  // set pin 2 on L293D high
    digitalWrite(motor1Pin2, LOW);   // set pin 7 on L293D low
	digitalWrite(motor1Pin1, HIGH);  // set pin 10 on L293D high
    digitalWrite(motor1Pin2, LOW);   // set pin 15 on L293D low
  }
}
